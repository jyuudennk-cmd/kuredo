<script>
    const allData = [
        { cat: "お出迎え", q: "「いらっしゃいませ」に○○と「○○」を添えたお出迎えをいたします。", a: "「いらっしゃいませ」に<b>笑顔</b>と<b>「一言」</b>を添えたお出迎えをいたします。" },
        { cat: "お声掛け", q: "声を掛けられたら○○、○○○からお声掛けをいたします。", a: "声を掛けられたら<b>負け</b>、<b>従業員</b>からお声掛けをいたします。" },
        { cat: "提案", q: "お客様のお困りごとを解決する○○○○○○○○をいたします。", a: "お客様のお困りごとを解決する<b>フィットコンサル</b>をいたします。" },
        { cat: "アフター", q: "○○をお渡しし、使用後の○○まで、○○をもって対応をいたします。", a: "<b>名刺</b>をお渡しし、使用後の<b>満足</b>まで、<b>責任</b>をもって対応をいたします。" },
        { cat: "お見送り", q: "送り出し時、お客様の「○○○」でお見送りをいたします。", a: "送り出し時、お客様の「<b>お名前</b>」でお見送りをいたします。" }
    ];

    let learnedIndexes = JSON.parse(localStorage.getItem('credoProgress')) || [];
    let currentIndex = 0;

    function updateDisplay() {
        const remaining = allData.filter((_, i) => !learnedIndexes.includes(i));
        const progress = (learnedIndexes.length / allData.length) * 100;
        document.getElementById('progress-bar').style.width = progress + '%';
        document.getElementById('stat-text').textContent = `習得済み: ${learnedIndexes.length} / 全${allData.length}問`;

        if (remaining.length === 0) {
            document.getElementById('cat-display').textContent = "祝";
            document.getElementById('q-display').textContent = "すべてのクレドをマスターしました！";
            document.getElementById('a-display').textContent = "今日もおもてなしの心で、最高の一日を。";
            return;
        }

        // 習得済みを避けて表示
        while (learnedIndexes.includes(currentIndex)) {
            currentIndex = (currentIndex + 1) % allData.length;
        }

        const data = allData[currentIndex];
        document.getElementById('cat-display').textContent = data.cat;
        document.getElementById('q-display').textContent = data.q;
        document.getElementById('a-display').innerHTML = data.a;
        document.getElementById('card').classList.remove('is-flipped');
    }

    function markAsLearned() {
        if (!learnedIndexes.includes(currentIndex)) {
            learnedIndexes.push(currentIndex);
            localStorage.setItem('credoProgress', JSON.stringify(learnedIndexes));
            nextCard();
        }
    }

    function nextCard() {
        currentIndex = (currentIndex + 1) % allData.length;
        updateDisplay();
    }

    function resetProgress() {
        if (confirm('進捗をリセットして最初から学習しますか？')) {
            learnedIndexes = [];
            localStorage.removeItem('credoProgress');
            currentIndex = 0;
            updateDisplay();
        }
    }

    // 初回表示
    updateDisplay();
</script>
